me_illusionist.1 = {
    type = country_event

    title = me_illusionist_1_title
    desc = me_illusionist_1_desc
    picture = city_country
    left_portrait = current_ruler
    left_portrait = scope:random_cleric_council
    right_portrait = scope:random_cleric_council_2
    right_portrait = scope:random_cleric_council_3

    immediate = {
        hidden_effect = {
            c:EGA= {
                create_character = {
                    first_name = "Srampun"
                    age = 26
                    female = no
                    culture = eagelian
                    religion = the_first_emperor
                    save_scope_as = random_cleric_council
                }
                scope:random_cleric_council = {
                    add_trait = cleric_trait
                }
            }
            c:LF0= {
                create_character = {
                    first_name = "Tialawen"
                    age = 40
                    female = no
                    culture = waldonian
                    religion = mother_earth_pantheon
                    save_scope_as = random_cleric_council_2
                }
                scope:random_cleric_council_2 = {
                    add_trait = illusionist_trait
                }
            }
            c:WGA= {
                create_character = {
                    first_name = "Eszurepesun"
                    age = 76
                    female = no
                    culture = weagelian
                    religion = the_first_emperor
                    save_scope_as = random_cleric_council_3
                }
                scope:random_cleric_council_2 = {
                    add_trait = archmage_trait
                }
            }
        }
    }

    option = {
        name = "me_illusionist_1.a"
        custom_tooltip = me_illusionist_1_tooltip
    }
}
