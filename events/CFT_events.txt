hidden_events.9 = {
    type = character_event
    hidden = yes

    immediate = {
        capital_scope = {
            if = {
                limit = {
                    any_pops_in_province = {
                        OR = {
                            pop_type = citizen
                            pop_type = freemen
                            pop_type = tribesmen
                        }
                    }
                }
                while = {
                    count = 5
                    limit = {
                        any_pops_in_province = {
                            OR = {
                                pop_type = citizen
                                pop_type = freemen
                                pop_type = tribesmen
                            }
                        }
                    }
                    random_pops_in_province = {
                        limit = {
                            OR = {
                                pop_type = citizen
                                pop_type = freemen
                                pop_type = tribesmen
                            }
                        }
                        set_pop_religion = root.religion
                    }
                }
            }
        }
    }
}
