namespace = deva_and_pros


#devastation
deva_and_pros.1 = {
	type = country_event
	title = "provschk_choo.1"
	desc = "provschk_choo.1"
	picture = revolt
	hidden = yes
	
	trigger = {
		any_owned_province = {
			OR = {
				NOT = { controller = ROOT }
				has_variable = in_plague
				has_siege = yes
			}		
		}
	}

	immediate = {  
		every_owned_province = {  		
			limit = { 
				OR = { 
					NOT = { controller = ROOT } 
					has_variable = in_plague 
					has_siege = yes 
				}  
			}									
			change_devastation_level = { VALUE = devastation_inverse_growth } 			
			check_is_devastation_at_proper_level = yes
			assing_proper_devastation_modifier = yes			
			decrease_prosperity_if_deva = { VALUE = -1 }
			if = { limit = { var:devastation_level > 10 } decrease_prosperity_if_deva = { VALUE = -1 } }
			if = { limit = { var:devastation_level > 20 } decrease_prosperity_if_deva = { VALUE = -1 } }
			if = { limit = { var:devastation_level > 30 } decrease_prosperity_if_deva = { VALUE = -1 } }
			if = { limit = { var:devastation_level > 40 } decrease_prosperity_if_deva = { VALUE = -1 } }
			check_is_prosperity_at_proper_level = yes
		}	
	}	
}

deva_and_pros.2 = {
	type = country_event
	title = "provschk_choo.1"
	desc = "provschk_choo.1"
	picture = revolt
	hidden = yes
	
	trigger = {
		has_war_exhaustion < 2	 
	}

	immediate = {
		every_owned_province = {
			limit = {  
				NOT = { has_variable = in_plague } 
				controller = ROOT  
				has_siege = no  
			}
			if = { 
				limit = { 
					has_variable = devastation_level 
					has_variable = prosperity_level 
				} 
				change_devastation_level = { VALUE = devastation_inverse_decay } 		
				assing_proper_devastation_modifier = yes		
				if = { 
					limit = { 
						owner = { war = no } 
					} 
					random_list = { 
						12 = { increase_prosperity_no_deva = { VALUE = 0.5 } }
						3 = { increase_prosperity_no_deva = { VALUE = 1 } }
						85 = {}
					}
				}
				check_is_prosperity_at_proper_level = yes
				check_is_devastation_at_proper_level = yes
				assing_proper_prosperity_modifier = yes						
			}
			else = {
				set_variable = { name = prosperity_level value = 0  }
				set_variable = { name = devastation_level value = 0  }		
			}		
		}
	}	
}
