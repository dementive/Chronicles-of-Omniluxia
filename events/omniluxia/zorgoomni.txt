me_zorgoomni.5 = {
    type = country_event

    title = me_zorgoomni.5_title
    desc = me_zorgoomni.5_desc
    picture = diadochi_coronation
    left_portrait = current_ruler
    right_portrait = scope:tadm_mage_scope

    fire_only_once = yes

    immediate = {
        hidden_effect = {
            create_character = {
                    first_name = "Tadm"
                    age = 26
                    female = no
                    culture = root.culture
                    religion = root.religion
                    save_scope_as = tadm_mage_scope
                    
                }
                scope:tadm_mage_scope = {
                    add_trait = cultured
                    add_trait = brave
                    add_trait = mage_trait
                    add_trait = ambitious
                }
            }
        }

    option = {
        name = "me_zorgoomni.5.a"
        custom_tooltip = me_zorgoomni.5_tooltip
        add_treasury = {
            add = root.has_monthly_income
            multiply = 6
        }
    }
    option = {
        name = "me_zorgoomni.5.b"
        custom_tooltip = me_zorgoomni.5_tooltip
        add_treasury = {
            subtract = root.has_monthly_income
            multiply = 3
        }
        current_ruler = {
            add_popularity = popularity_medium
        }
    }
    option = {
        highlight = yes
        name = "me_zorgoomni.5.c"
        custom_tooltip = me_zorgoomni.5_tooltip
        scope:tadm_mage_scope = {
           death = {
                death_reason = death_stabbing
                killer = root.current_ruler
            }
        }
        current_ruler = {
            add_trait = harsh
        }
    }
}

}
