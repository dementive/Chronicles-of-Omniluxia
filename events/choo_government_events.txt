namespace = choo_governments

#tolerance/intolerance mini mechanic
choo_governments.1 = {
	type = country_event
	title = choo_governments.1.t

	trigger = {
			is_monarchy =  yes
		OR = {
			
			has_law = tolerance_for_the_other
			has_law = religious_persecutions
			has_law = monastic_expulsion_of_heatens

		}
	}


	immediate = {
		if = {
		limit = {  has_law = tolerance_for_the_other }	
		
		}
		else_if = {
			limit = { 
				OR = { 
				has_law = religious_persecutions 
				has_law = monastic_expulsion_of_heatens 
				} 
			}

		}
		
	}
		
	option = {
		ai_chance = {
			factor = 100
		}
	}

}
