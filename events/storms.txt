namespace = storm

#1-10 Sea Storms
#These are being spawned once a year "from" a random country. This means sometimes there will be no storms, if that country is destroyed before they break out.

#Sea Storms
storm.1 = { #Sea Storms
	type = country_event
	hidden = yes

	immediate = {
		while = {
			count = 3
			random_sea_and_river_zone = {
				limit = {
					terrain = ocean
					NOT = { has_province_modifier = stormy_seas }
				}
				add_vfx = {
					name = "seastorm"
					duration = 60
				}
				add_province_modifier = {
					name = "stormy_seas"
					duration = 60
				}
				every_neighbor_province = {
					limit = {
						terrain = ocean
						NOT = { has_province_modifier = stormy_seas }
					}
					add_vfx = {
						name = "seastorm"
						duration = 60
					}
					add_province_modifier = {
						name = "stormy_seas"
						duration = 60
					}
				}
			}
		}
	}
}

storm.3 = { #Snowstorms
	type = country_event
	hidden = yes
	immediate = {
		random_province = {
			limit = {
				OR = {
					has_province_modifier = frigid_climate
					has_province_modifier = alpine_climate
				}
				has_winter = yes
				NOT = { has_province_modifier = snow_storm }
			}
			add_vfx = {
				name = "snowstorm"
				duration = 60
			}
			add_province_modifier = {
				name = "snow_storm"
				duration = 60
			}
			every_neighbor_province = {
				limit = {
					OR = {
						has_province_modifier = frigid_climate
						has_province_modifier = alpine_climate

}
