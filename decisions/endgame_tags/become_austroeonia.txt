	become_austroeonia = {
		
		potential = {
			num_of_cities >= 1
			primary_culture = austroeonian
		}
		
		highlight = {
			scope:province = {
				OR = {
					is_in_region = austropetolian_isle
					is_in_region = austropetolian_archipelago
				}
			}
		}
		
		allow = {
			trigger_if = {
				limit = {
					any_country = {
						tag = ASA
					}
				}
				custom_tooltip = {
					text = formable_not_austroeonia_exists
					NOT = {
						any_country = {
							tag = ASA
						}
					}
				}
			}
			is_subject = no
			can_form_nation_trigger = yes
			owns_or_subject_owns_region = austropetolian_isle
			owns_or_subject_owns_region = austropetolian_archipelago
		}
		
		
		effect = {
			custom_tooltip = "tier_3_formable_tooltip"
			if = {
				limit = {
					NOR = { 
						government = imperium 
						government = imperial_cult
					}
				}
				republic_to_monarchy_law_variable_effect = yes # decide which legion law should be transferred
				change_government = imperium
				republic_to_monarchy_law_change_effect = yes # change legion law based on variable if present
				switch_government_type_event_clearup_effect = yes
			}
			change_country_name = "ASA"
			hidden_effect = {
				change_country_adjective = "ASAADJ"
				change_country_tag = ASA
			}
			add_5_free_province_investments = yes
			capital_scope = {
				capital_formable_large_effect = yes
				formable_capital_modifier_large_effect = yes
			}
			add_country_modifier = {
				name = formable_country_huge_modifier
				duration = 7600
			}
		}
		
		ai_will_do = {
			base = 1
		}
	}

}
