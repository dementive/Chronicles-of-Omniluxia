country_decisions = {
	
	#form Nouveau Aralan
	form_nouveau_aralan = {
	
		potential = {
			num_of_cities >= 5
			country_culture_group = aralans
			NOT = {
				tag = NAR
			}
			OR = {
				is_ai = no
				num_of_cities >= 10
			}
			NOR = {
				is_tier_1_formable_trigger = yes
				is_tier_2_formable_trigger = yes
				is_endgame_tag_trigger = yes
			}
		}
		
		highlight = {
			scope:province = {		
			OR = {
				is_in_area = rohevia_area_1
				is_in_area = rohevia_area_2
				is_in_area = rohevia_area_3		
				is_in_area = rohenoa_shout_area_1	
				is_in_area = rohenoa_shout_area_2				
				}
			}
		}
		
		allow = {
			custom_tooltip = {
				text = formable_not_nouveau_aralan_exists
				NOT = {
					any_country = {
						tag = NAR
					}
				}
			}
			can_form_nation_trigger = yes
		}
		
		ai_allow = {
			area:rohevia_area_1 = {
				any_area_province = {
					count = all
					owned_or_subject_owned = root
				}
			}
			area:rohevia_area_2 = {
				any_area_province = {
					count = all
					owned_or_subject_owned = root
				}
			}
			area:rohevia_area_3 = {
				any_area_province = {
					count = all
					owned_or_subject_owned = root
				}
			}
			area:rohenoa_shout_area_1 = {
				any_area_province = {
					count = all
					owned_or_subject_owned = root
				}
			}			
			area:rohenoa_shout_area_2 = {
				any_area_province = {
					count = all
					owned_or_subject_owned = root
				}
			}
		}

		
		effect = {
			custom_tooltip = "tier_1_formable_tooltip"		
			change_country_name = "NAR"
			hidden_effect = {
				change_country_adjective = "NAR_ADJ"
				change_country_color =  "dark_burgundy"
				change_country_tag = NAR
				change_country_flag = "NAR_FLAG"
			
				every_owned_province = {
					every_pops_in_province = {
						limit = {
							pop_culture_group = aralans
						}
						set_pop_culture = aralan_rohenoan
					}
				}
				every_province = {
					limit = {
						OR = {
							is_in_area = rohevia_area_1
							is_in_area = rohevia_area_2
							is_in_area = rohevia_area_3
							is_in_area = rohenoa_shout_area_1
							is_in_area = rohenoa_shout_area_2							
						}
						NOT = { owner = ROOT }
					}
					add_claim = ROOT
				}
			}
			add_2_free_province_investments = yes

			change_government = feudal_monarchy
			set_primary_culture = aralan_rohenoan
			add_3_free_province_investments = yes
		}
		
		
		ai_will_do = {
			base = 1
		}
	}
	
}
