country_decisions = {
	
	#form Nouveau Aralan
	form_nouveau_aralan = {
	
		potential = {
			num_of_cities >= 1
			this.culture.culture_group = culture_group:aralans
			NOT = {
				tag = NAR
			}
			NOR = {
				is_tier_1_formable_trigger = yes
				is_tier_2_formable_trigger = yes
				is_endgame_tag_trigger = yes
			}
		}
		
		highlight = {
			scope:province = {		
				OR = {
					is_in_region = rohevia_region
					is_in_region = rohenoa_shout_region			
				}
			}
		}
		
		allow = {
			custom_tooltip = {
				text = formable_not_nouveau_aralan_exists
				NOT = {
					any_country = {
						tag = NAR
					}
				}
			}
			can_form_nation_trigger = yes
			owns_or_subject_owns_area = rohevia_area_1
			owns_or_subject_owns_area = rohevia_area_2
			owns_or_subject_owns_area = rohevia_area_3
			owns_or_subject_owns_area = rohenoa_shout_area_1
			owns_or_subject_owns_area = rohenoa_shout_area_2
		}

		
		effect = {
			custom_tooltip = "tier_1_formable_tooltip"		
			change_country_name = "NAR"
			capital_scope = {
				establish_city_effect = yes
				capital_formable_medium_effect = yes
				formable_capital_modifier_normal_effect = yes
			}
			#Change up to 1,000 old culture pops to new culture
			hidden_effect = {
				every_owned_province = {
					while = {
						count = 1000
						every_pops_in_province = {
							limit = {
								culture = root.culture
							}
							set_pop_culture = aralan_rohenoan
						}
					}
				}
				every_character = {
					limit = {
						AND = {
							OR = {
								from_ruler_family = yes
								is_minor_character = no
							}
							has_culture = root.culture
						}
					}
					set_culture = aralan_rohenoan
				}
				set_primary_culture_cleanup_effect = {
					NEW_PRIMARY_CULTURE = aralan_rohenoan
					MAKE_OLD_CULTURE_INTEGRATED = yes
				}
				change_country_adjective = "NAR_ADJ"
				change_country_color =  "dark_burgundy"
				change_country_tag = NAR
				change_country_flag = "NAR_FLAG"

				region:rohevia_region = {
					every_region_province = {
						add_claim = root
					}
				}
				region:rohenoa_shout_region = {
					every_region_province = {
						add_claim = root
					}
				}
				region:rohenoa_north_region = {
					every_region_province = {
						add_claim = root
					}
				}
				region:rohenoa_central_region = {
					every_region_province = {
						add_claim = root
					}
				}
			}
			add_2_free_province_investments = yes
			monarchy_formable_government_change_effect = yes
			change_government = feudal_monarchy
			set_primary_culture = aralan_rohenoan
			add_3_free_province_investments = yes
		}
		
		
		ai_will_do = {
			base = 1
		}
	}
	
}
