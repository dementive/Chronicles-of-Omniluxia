#General use#
char_has_no_items = {
	scope = character

	is_shown = { 

		has_any_item = no

	}

}



#specific for items#

##equip##
#normal item_sword
equip_item_sword = {
	
	is_valid = { 
		
		

		}

		effect = {
			if = {

				limit = {  has_character_modifier = item_sword_modifer  }
				
				add_character_modifier = {
					name = item_sword_modifer
					duration = -1
				}
	
			}
			else = {

				remove_character_modifier = item_sword_modifer

			}


		}



}





#discarding#
discard_item_sword = {
	is_shown = { 
	


	}
	is_valid = { 
		
		has_item = { item_ID = item_sword }

		}
		effect = {
		
		remove_character_item = { item_ID = item_sword  }

		}



}
