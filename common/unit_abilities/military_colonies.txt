military_colonies = {

	duration = 0
	enable = no
	toggle = no
	army_only = yes
	
	map = yes
	soundeffect ="event:/SFX/Animations/Units/Generic/sfx_anim_unit_build_colony"
	
	
	hidden = {
        unit_owner = { 
            NOR = { 
               	is_tradition_tree_allowed = golden_dwarf_philosophy
               	is_tradition_tree_allowed = high_elven_philosophy
               	is_tradition_tree_allowed = levantine_philosophy
               	is_tradition_tree_allowed = northerner_philosophy
               	is_tradition_tree_allowed = zarakens_philosophy
               	is_tradition_tree_allowed = zinduidian_philosophy
            } 
        }
	}

	allow = {
		unit_owner = { 
			trigger_if = {
				limit = { is_tradition_tree_allowed = golden_dwarf_philosophy }
				has_military_bonus = larsa_dwarf_path_4
			}
			trigger_else_if = {
				limit = { is_tradition_tree_allowed = high_elven_philosophy }
				has_military_bonus = high_elf_colonial_path_2
			}
			trigger_else_if = {
				limit = { is_tradition_tree_allowed = levantine_philosophy }
				has_military_bonus = levantine_egyptian_path_4
			}
			trigger_else_if = {
				limit = { is_tradition_tree_allowed = northerner_philosophy }
				has_military_bonus = norrfallian_seafarers_path_5
			}
			trigger_else_if = {
				limit = { is_tradition_tree_allowed = zarakens_philosophy }
				has_military_bonus = zarakens_katssaspion_path_4
			}
			trigger_else_if = {
				limit = { is_tradition_tree_allowed = zinduidian_philosophy }
				has_military_bonus = zinduidian_pirate_path_4
			}
		}
		has_commander = yes
		is_army = yes
		in_combat = no
		is_moving = no
		in_siege = no
		is_exiled = no
		unit_size >= 5
		custom_tooltip = {
			text = "num_of_loyal_cohorts_tooltip"
			num_of_non_loyal_cohorts_svalue >= 1
		}
		hidden:unit_owner = { can_pay_price = create_military_colony }
		hidden:unit_location = {
			has_owner = yes
			total_population <= 5
			state_loyalty >= 33
			custom_tooltip = {
				text = military_colonies_capital_tooltip
				NOT = {
					state = {
						any_state_province = {
							is_capital = yes
						}
					}
				}
			}
			owner = root.unit_owner 
		}
		trigger_if = {
			limit = {
				exists = commander
			}
			commander = {
				loyalty > 33
			}
		}
	}
	
	
	start_effect = {
		unit_owner = {  pay_price = create_military_colony }
		if = {
			limit = {
				exists = commander
			}
			random_sub_unit = {
				limit = {
					has_personal_loyalty = no
				}
				if = {
					limit = {
						root = {
							unit_owner = {
								is_republic = yes
							}
						}
					}
					set_personal_loyalty = root.commander
				}
				else_if = {
					limit = {
						root = {
							commander = {
								is_ruler = no
							}
						} 
					}
					set_personal_loyalty = root.commander
				}
				else = {
					root = {
						unit_owner = {
							add_tyranny = 1
						}
					}
				}
			}
		}
		unit_location = {
			create_state_pop = freemen
			add_state_loyalty = -15
		}
	}
}
