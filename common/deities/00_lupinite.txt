	trigger = {
		polytheistic_check_religion_trigger = yes
		OR = {
			very_rare_deity_trigger = {
				RELIGION = great_wolf_pantheon
			}
			deity:omen_lula = {
				holy_site_deity_check_trigger = yes
			}
		}
	}
	icon = deity_war

	deification_trigger = {
		can_deify_trigger = yes
	}

	}
}
