save_adventure_companion_scopes = {
	random_character = {
		limit = {
			is_friend = root
			is_adult = yes
		}
		save_scope_as = adventurer_1
	}
	if = {
		limit = {
			NOT = { exists = scope:adventurer_1 }
		}
		save_adventurer = {
			SCOPE = "adventurer_1"
			LIMIT = ""
		}
	}
	random_character = {
		limit = {
			is_friend = root
			is_adult = yes
			NOT = { THIS = scope:adventurer_1 }
		}
		save_scope_as = adventurer_2
	}
	if = {
		limit = {
			NOT = { exists = scope:adventurer_1 }
		}
		save_adventurer = {
			SCOPE = "adventurer_2"
			LIMIT = "NOT = { THIS = scope:adventurer_1 }"
		}
	}
	save_adventurer = {
		SCOPE = "adventurer_3"
		LIMIT = "NOR = { THIS = scope:adventurer_1 THIS = scope:adventurer_2 }"
	}
	save_adventurer = {
		SCOPE = "adventurer_4"
		LIMIT = "NOR = { THIS = scope:adventurer_1 THIS = scope:adventurer_2 THIS = scope:adventurer_3 }"
	}
	save_adventurer = {
		SCOPE = "adventurer_5"
		LIMIT = "NOR = { THIS = scope:adventurer_1 THIS = scope:adventurer_2 THIS = scope:adventurer_3 THIS = scope:adventurer_4 }"
	}
	save_adventurer = {
		SCOPE = "adventurer_6"
		LIMIT = "NOR = { THIS = scope:adventurer_1 THIS = scope:adventurer_2 THIS = scope:adventurer_3 THIS = scope:adventurer_4 THIS = scope:adventurer_5 }"
	}
}

save_adventurer = {
	random_character = {
		limit = {
			$LIMIT$
			trigger_if = {
				limit = {
					gender_equality = no
				}
				is_male = yes
			}
			is_adult = yes
		}
		weight = {
			modifier = {
				factor = 25
				corruption > 20
			}
			modifier = {
				factor = 25
				prominence >= 70
			}
			modifier = {
				factor = 50
				has_same_family = root
			}
		}
		save_scope_as = $SCOPE$
	}
}

save_adventurer_var = {
	if = {
		limit = {
			NOT = { has_variable = adventurer_1 }
		}
		set_variable =  {
			name = adventurer_1
			value = scope:$SCOPE$
		}
	}
	else_if = {
		limit = {
			NOT = { has_variable = adventurer_2 }
		}
		set_variable =  {
			name = adventurer_2
			value = scope:$SCOPE$
		}
	}
	else_if = {
		limit = {
			NOT = { has_variable = adventurer_3 }
		}
		set_variable =  {
			name = adventurer_3
			value = scope:$SCOPE$
		}
	}
}
