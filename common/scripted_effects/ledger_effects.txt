sort_territories = {

	every_in_list = {
		variable = territories

		root = {
			add_to_variable_list = {
				name = territories_temp
				target = prev
			}
		}
	}

	clear_variable_list = territories

	ordered_in_list = {
		variable = territories_temp
		order_by = $sorting$
		min = 0

		root = {
			add_to_variable_list = {
				name = territories
				target = prev
			}
		}
	}

	set_variable = {
		name = sorting
		value = $num$
	}
}
