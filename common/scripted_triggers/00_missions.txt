
has_tendency_to_tyranny_trigger = {
	OR = {
		has_trait = ambitious
		has_trait = selfish
		has_trait = jealous
		has_trait = vengeful
		has_trait = righteous
		has_trait = arbitrary
		is_friend = root.current_ruler
    }
}

food_trade_good_trigger = {
    OR = {
        trade_goods = grain
        trade_goods = vegetables
        trade_goods = fish
        trade_goods = salt
        trade_goods = cattle
        trade_goods = meat
        trade_goods = cheese
		trade_goods = honey
    }
}

valuable_trade_good_trigger = {
    OR = {
        trade_goods = papyrus
        trade_goods = cloth
        trade_goods = earthware
        trade_goods = marble
        trade_goods = incense
        trade_goods = gems
    }
}

military_trade_good_trigger = {
    OR = {
        trade_goods = camel
        trade_goods = horses
        trade_goods = wood
        trade_goods = hardwood
    }
}

resource_building_trigger = {
	trigger_if = {
		limit = {
			has_minable_trade_good_trigger = yes
		}
		has_building = slave_mine_building
	}
	trigger_else_if = {
		limit = {
			has_food_trade_good_trigger = yes
		}
		has_building = basic_settlement_infratructure_building
	}
	trigger_else = {
		has_building = latifundia_building
	}
}

	trigger_if = {
