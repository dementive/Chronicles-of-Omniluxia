	}

			$TIME$ = 2
		}

	}

	}
}

trade_good_decision_allow = {
	#INPUT: DIFFICULTY = X (1-4)
	trigger_if = {
		limit = {
			$DIFFICULTY$ = 1
		}
		has_city_status = yes
		num_goods_produced >= 2
		total_population >= 20
	}
	trigger_else_if = {
		limit = {
			$DIFFICULTY$ = 2
		}
		has_city_status = yes
		total_population >= 30
		num_goods_produced >= 2
	}
	trigger_else_if = {
		limit = {
			$DIFFICULTY$ = 3
		}
		has_city_status = yes
		total_population >= 40
		num_goods_produced >= 3
		free_building_slots = 0
		has_no_ongoing_construction = yes
	}
	trigger_else_if = {
		limit = {
			$DIFFICULTY$ = 4
		}
		has_city_status = yes
		total_population >= 50
		num_goods_produced >= 4
		free_building_slots = 0
		has_no_ongoing_construction = yes
	}
	trigger_else = {
		has_city_status = yes
		num_goods_produced >= 2
		total_population >= 20
	}
}
