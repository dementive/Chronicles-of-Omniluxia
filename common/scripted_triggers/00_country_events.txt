		NOT = {
			any_character = { has_variable = ip_rep_66_supporter_flag }
		}

			has_variable = ip_rep_66_supporter_flag
			save_temporary_scope_as = ip_rep_66_supporter_1_temp
		}
		AND = {
			any_character = { 
				has_variable = ip_rep_66_supporter_flag
				save_temporary_scope_as = ip_rep_66_supporter_1_temp
			}
			any_character = { 
				has_variable = ip_rep_66_supporter_flag
				NOT = { this = scope:ip_rep_66_supporter_1_temp }
				save_temporary_scope_as = ip_rep_66_supporter_2_temp

		}

		NOT = {
			any_character = { has_variable = ip_rep_66_counter_flag }

			}

	}
