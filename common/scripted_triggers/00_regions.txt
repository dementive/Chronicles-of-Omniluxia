# Volcano Triggers
country_borders_volcano_province_trigger = {
	any_owned_province = {
		OR = {
			city_borders_orc_volcano_trigger = yes
			city_borders_snake_volcano_trigger = yes
		}
	}
}

country_borders_volcano_province_not_erupted_trigger = {
	any_owned_province = {
		OR = {
			city_borders_orc_volcano_trigger = yes
			city_borders_snake_volcano_trigger = yes
		}
		NOT = { has_variable = recent_volcano_eruption }
	}
}

city_borders_orc_volcano_trigger = {
	any_neighbor_province = {
		province_id = 5789
	}
}

city_borders_snake_volcano_trigger = {
	any_neighbor_province = {
		province_id = 5790
	}
}
