has_item = {

    is_target_in_variable_list = {  
        name = item_list   
        target = flag:$item_ID$
                                }
}
has_any_item = {

    has_variable_list = item_list

}

has_equiped_item = {

    is_target_in_variable_list = {  
        name = item_list   
        target = flag:$item_ID$
                                }

    has_character_modifier = $item_ID$_modifier

}
