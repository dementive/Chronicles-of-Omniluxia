##### Cultural groups for flags

# has_barbarian_flag_trigger
# has_celtic_flag_trigger

has_barbarian_flag_trigger = {
	OR = {
		always = no
	}
}
has_celtic_flag_trigger = {
	OR = {
		always = no
	}
}


has_white_flag_trigger = {
	ROOT.color.saturation <= 0.2
	ROOT.color.brightness >= 0.5
}

has_yellow_flag_trigger = {
	ROOT.color.hue >= 0.10
	ROOT.color.hue <= 0.17
	ROOT.color.saturation >= 0.2
	ROOT.color.brightness >= 0.2
}
