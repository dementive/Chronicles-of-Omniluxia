ruler_retire = { 

	on_other_nation = no
	on_own_nation = yes
	
	sound = "event:/SFX/UI/Character/Unique/sfx_ui_character_arrange_marriage"
	
	message = yes
	
	potential_trigger = {
		scope:actor = {
			is_ai = no
			is_republic = no
			#To important tags from retiring early
			trigger_if = {
				limit = {
					OR = {
						tag = EGA
						tag = WGA
						tag = GAE
					}
				}
				current_date > 1051.1.1
			}
		}
		hidden:scope:target = {
			is_alive = yes
			is_ruler = yes
			OR = {
				age > 60
				AND = {
					health < 15
					age > 50
				}
			}
			employer = scope:actor
			NOT = { employer = { tag = REB } }
			NOT = { employer = { tag = BAR } }
		}
	}
	
	allowed_trigger = {
		scope:actor = {
			can_pay_price = ruler_retire_price
			possible_holdings > 1
		}
		scope:target = {

		}
	}
	
	#for any province in the scope:actor country (stolen from get_holding)
	province_actor_trigger = {
		OR = {
			NOT = {
				exists = holding_owner
			}
			AND = {
				exists = holding_owner
				holding_owner = {
					is_alive = no
				}
			}
		}
		NOT = {
			this = scope:actor.capital_scope
		}
	}
	
	effect = {
		scope:actor = {
			pay_price = ruler_retire_price
			set_as_ruler = primary_heir
		}

		scope:target = {
			hidden_effect = { remove_all_offices = yes }
			add_character_modifier = {
				name = retired_ruler
				duration = -1
			}
			if = {	
				limit = {
					exists = scope:recipient
				}
				scope:target = {
					add_holding = scope:recipient
				}
			}			
		}
	}	
}
