increase_popularity = {

	on_other_nation = no
	on_own_nation = yes
	
	message = no
	
	sound = "event:/SFX/UI/Character/Generic/sfx_ui_character_scholar"
	
	potential_trigger = {
		hidden:scope:target = {
			popularity < 95 #Can't increase it above 95
			employer = scope:actor
			is_ruler = yes
			is_alive = yes
			employer = {
				OR = {
					is_monarchy = yes
					is_republic = yes
				}
			}
			NOT = { employer = { tag = REB } }
			NOT = { employer = { tag = BAR } }
		}
	}
	
	allowed_trigger = {
		scope:actor = {
			trigger_if = {
				limit = { has_variable = popularity_cooldown }
				custom_tooltip = {
					text = popularity_cooldown_tt
					NOT = { has_variable = popularity_cooldown }
				}
			}
		}
		scope:target = {
			trigger_if = {
				limit = { is_adult = no }
				is_adult = yes
			}
			is_ruler = yes
			wealth >= 300
			NOT = {
				has_ambition = scheme_ruler_popularity
			}
		}
	}

	effect = {
		scope:target = {
			clear_ambition_effect_ci = yes
			set_variable = {
				name = popularity_cooldown
				days = 1825
			}
			set_variable = {
				name = ongoing_scheme_target
				value = scope:target
				days = 770
			} 
			set_ambition = scheme_ruler_popularity
		}
		custom_tooltip = popularity_ambition_effect_tt
	}
	
	ai_will_do = {
		modifier = {
			add = 5
			trigger = {
				is_ruler = yes
				popularity < 15
			}
		}
	}
	
}
