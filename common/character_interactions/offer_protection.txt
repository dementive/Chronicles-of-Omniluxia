offer_protection = {

	on_other_nation = yes
	on_own_nation = no
	
	message = yes
	
	sound = "event:/SFX/UI/Character/Generic/sfx_ui_character_corruption"
	
	potential_trigger = {
		hidden:scope:target = {
			NOT = { employer = scope:actor }
			NOT = { employer = { tag = REB } }
			NOT = { employer = { tag = BAR } }
			employer = {
				#num_of_cities < 20
				war = no
				is_subject = no
				any_neighbour_country = {
					THIS = scope:actor
				}
				is_overlord = no
			}
			is_alive = yes
			is_ruler = yes
		}
		scope:actor = {
			is_subject = no
			invention = intervene_inv
		}
	}
	
	
	allowed_trigger = {
		scope:actor = {
			can_pay_price = offer_protection_price
			trigger_if = {
				limit = { has_variable = protection_cooldown }
				custom_tooltip = {
					text = protection_cooldown_tt
					NOT = { has_variable = protection_cooldown }
				}
			}
		}
		scope:target = {
			is_friend = scope:actor.current_ruler
			employer = {
				opinion = {
					target = scope:actor
					value >= 50
				}
			}
		}
	} 

	effect = {
		scope:actor = {
			set_variable = {
				name = protection_cooldown
				days = 3650
			}
			pay_price = offer_protection_price
			trigger_event = {
				id = character_interactions_inv.6
				days = 5
			}
		}
	}
}
