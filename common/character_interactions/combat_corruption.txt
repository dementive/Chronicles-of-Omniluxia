combat_corruption = {

	on_other_nation = no
	on_own_nation = yes
	
	message = no
	
	sound = "event:/SFX/UI/Character/Generic/sfx_ui_character_scholar"
	
	potential_trigger = {
		hidden:scope:target = {
			employer = scope:actor
			is_ruler = yes
			is_alive = yes
			employer = {
				OR = {
					is_monarchy = yes
					is_republic = yes
				}
			}
			corruption >= 50 #Only for really corrupt bitches
			NOT = { employer = { tag = REB } }
			NOT = { employer = { tag = BAR } }
		}
	}
	
	allowed_trigger = {
		scope:actor = {
			trigger_if = {
				limit = { has_variable = corruption_cooldown }
				custom_tooltip = {
					text = corruption_cooldown_tt
					NOT = { has_variable = corruption_cooldown }
				}
			}
		}
		scope:target = {
			trigger_if = {
				limit = { is_adult = no }
				is_adult = yes
			}
			is_ruler = yes
			wealth >= 300
			NOT = {
				has_ambition = scheme_ruler_corruption
			}
		}
	}

	effect = {
		scope:target = {
			set_variable = {
				name = corruption_cooldown
				days = 1825
			}
			clear_ambition_effect_ci = yes
			set_variable = {
				name = ongoing_scheme_target
				value = scope:target
				days = 770
			} 
			set_ambition = scheme_ruler_corruption
		}
		custom_tooltip = corruption_ambition_effect_tt
	}
	
	ai_will_do = {
		modifier = {
			add = 5
			trigger = {
				is_ruler = yes
				corruption > 90
			}
		}
	}
}
