bribe_character = {

	on_other_nation = no
	on_own_nation = yes
	
	sound = "event:/SFX/UI/Character/Unique/sfx_ui_character_bribe"
	
	potential_trigger = {
		hidden:scope:target = {
			employer = scope:actor
			NOT = { employer = { tag = REB } }
			NOT = { employer = { tag = BAR } }
			is_alive = yes
			is_ruler = no
			scope:actor.current_ruler = {
				NOR = {
					has_trait = orator
					has_trait = silver_tongued
					has_trait = cultured
				}
			}
		}
	}
	
	
	allowed_trigger = {
		scope:target = {
			prisoner = no
			is_adult = yes
			trigger_if = {
				limit = {
					has_variable = ca_character_bribed
				}
				custom_tooltip = {
					text = "BRIBED_RECENTLY"
			
					NOT = { has_variable = ca_character_bribed }
				}
			}
			NOR = {
				has_trait = righteous
				has_trait = honest
				has_trait = trustworthy
			}
			NAND = {
				employer = {
					is_monarchy = yes
				}
				is_co_ruler = yes
			}
		}
		custom_tooltip = {
			text = BRIBE_RULERWEALTH_TT
			scope:actor.current_ruler = {
				wealth >= 100
			}
		}
		trigger_if = {
			limit = {
				scope:actor = {
					is_ai = no
				}
			}
			scope:actor = {
				can_pay_price = bribe_pi
			}
		}
	}
	

	
	effect = {
		if = {
			limit = {
				scope:actor = {
					is_ai = no
				}
			}
			scope:actor = {
				pay_price = bribe_pi
			}
		}
		hidden_effect = {
			scope:actor.current_ruler = {
				if = {
					limit = { scope:actor = { is_ai = no } }
					add_corruption = 1
				}
			}
			scope:actor.current_ruler = {
				add_gold = -100
			}
		}
		custom_tooltip = bribe_actor_tt
		hidden_effect = {
			scope:target = { # Update TT if changing this!
				add_gold = 100
				add_loyalty = ci_bribe
				add_corruption = 5
				hidden_effect = {
					set_variable = {
						name = ca_character_bribed
						days = 1440
					}
				}
			}
		}
		custom_tooltip = bribe_target_tt
	}
	
	ai_will_do = {
		modifier = {
			add = 100
			trigger = {
				loyalty <= 45
				OR = {
					has_any_office = yes
					is_governor = yes
					power_base > 5
				}
				
			}
		}
		modifier = {
			add = 100
			trigger = {
				loyalty <= 45
				OR = {
					is_general = yes
					is_admiral = yes
				}
			}
		}
	}
	
}
