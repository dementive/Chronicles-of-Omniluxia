state_loyalty_aimod = {
	trigger = {
		AND = {
			OR = {
				war = no
				has_aggressive_expansion > 50
			}
			OR = {
				has_aggressive_expansion > 40
				AND = {
					any_country_state = {
						percent > 0.15
						state_level_loyalty < 75
					}
					has_aggressive_expansion > 20
				}
				any_country_state = {
					percent > 0.15
					state_level_loyalty < 20
				}
			}
		}
	}
	aggressive = -80
	ae_ceiling = -15
	trustworthy = 10
}
